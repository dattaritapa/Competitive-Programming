class Solution {
public:
    int maxProfit(vector<int>& prices) {
        ios_base::sync_with_stdio(false);
        cin.tie(NULL);
        int size=prices.size();
        int dif=0;
        for(int i=1;i<size;i++)
        {
            if(prices[i]>prices[i-1])
                dif=dif+prices[i]-prices[i-1];
        }
        return dif;
    }
};
